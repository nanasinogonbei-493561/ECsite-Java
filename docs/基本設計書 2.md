  price DECIMAL(10,2) NOT NULL,
  volume INT,

  description TEXT,
  image_path VARCHAR(255),
  stock_quantity INT NOT NULL DEFAULT 0,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS orders (
  id BIGINT AUTO_INCREMENT PRIMARY KEY,
  order_number VARCHAR(20) UNIQUE NOT NULL,
  customer_name VARCHAR(100) NOT NULL,
  customer_email VARCHAR(255) NOT NULL,
  customer_phone VARCHAR(20),
  delivery_address TEXT,
  total_amount DECIMAL(10,2) NOT NULL,
  status VARCHAR(20) NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS order_items (
  id BIGINT AUTO_INCREMENT PRIMARY KEY,
  order_id BIGINT NOT NULL,

  CONSTRAINT fk_oi_order FOREIGN KEY (order_id) REFERENCES orders(id),
  CONSTRAINT fk_oi_product FOREIGN KEY (product_id) REFERENCES products(id)

6. ドメイン・DTO 主要定義（抜粋/Java）
// entity/Product.java
@Entity
public class Product {
  @Id @GeneratedValue private Long id;
  @Column(nullable=false) private String name;
  private String brewery;
  @Column(nullable=false, precision=10, scale=2) private BigDecimal price;
  private Integer volume;
  private BigDecimal alcoholContent;
  @Column(columnDefinition="TEXT") private String description;
  private String imagePath;
  @Column(nullable=false) private Integer stockQuantity;
  @CreationTimestamp private Instant createdAt;
  @UpdateTimestamp private Instant updatedAt;
}

// entity/Order.java
@Entity @Table(name="orders")
public class Order {
  @Id @GeneratedValue private Long id;
  @Column(unique=true, nullable=false) private String orderNumber;
  @Column(nullable=false) private String customerName;
  @Column(nullable=false) private String customerEmail;
  private String customerPhone;
  @Column(columnDefinition="TEXT") private String deliveryAddress;
  @Column(nullable=false, precision=10, scale=2) private BigDecimal totalAmount;
  @Column(nullable=false) private String status; // RECEIVED/SHIPPED/COMPLETED
  @CreationTimestamp private Instant createdAt;
}

// dto/OrderRequest.java
public record OrderRequest(
  List<Item> items,
  Customer customer
) {
  public record Item(Long productId, @Min(1) @Max(3) Integer quantity) {}
  public record Customer(
    @NotBlank String name,
    @Email String email,
    String phone,
    @NotBlank String deliveryAddress
  ) {}
}

7. ビジネスルール

年齢確認：初回アクセスで生年月日入力。>=20歳のみ継続利用可能（セッション）。

在庫：注文確定時に厳密チェック。OKなら stockQuantity -= quantity。

金額：MVPは「単価×数量」の合計のみ（送料/税は将来対応）。

注文番号：yyyy + 連番 or Base32ランダム（重複不可）。

ステータス遷移：RECEIVED → SHIPPED → COMPLETED のみ。逆戻り不可。

決済：代引きのみ。クレカ/振込はPhase2。

8. 例外・ログ方針
8.1 例外（共通ハンドラ）

@RestControllerAdviceにてMethodArgumentNotValidException等を400/422化

業務系（在庫/年齢/重複）は 409/422 を返す

8.2 構造化ログ（Logback）

形式（JSON推奨）：
timestamp, level, logger, message, traceId, path, user(agent/admin), orderNumber(optional)

重要イベント：ORDER_CREATED, ORDER_STATUS_CHANGED, ADMIN_LOGIN_SUCCEEDED/FAILED

9. セキュリティ

管理API：Spring Security + JWT（有効期限短め、HS256）

管理パスワード：BCrypt

CORS：フロントのオリジンに限定

画像アップロード：拡張子/サイズ/コンテントタイプ検証

ヘッダ：X-Content-Type-Options: nosniff, X-Frame-Options: DENY

年齢確認：クッキー/Session Storage（満了：ブラウザ閉じるまで）

  participant DB as DB

  User->>FE: 商品詳細を開く
  FE->>BE: GET /api/products/{id}
  BE->>DB: SELECT product
  DB-->>BE: product
  BE-->>FE: 200 JSON

14. 受け入れ条件（抜粋）

 年齢未確認の状態で商品ページに遷移すると年齢確認に誘導される

 在庫 0 の商品は「在庫なし」表示／注文不可

 数量は1〜3のみ受け付ける

 注文確定で在庫が減算され、注文番号が発行される

 管理者はログイン後に商品CRUD／注文ステータス更新が可能

 APIのエラーは規定フォーマットで返却される
